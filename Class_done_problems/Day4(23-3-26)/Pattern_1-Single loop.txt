import java.util.*;
public class Pattern_1_singleloop {
       public static void main(String[] args){
        Scanner sc = new Scanner(System.in);
           int n = sc.nextInt();
           sc.close();
           for(int i = 1; i <= n*n; i++){
              System.out.print("* ");
              if(i%n == 0){
               System.out.println();
              }
           }
        }
    }
