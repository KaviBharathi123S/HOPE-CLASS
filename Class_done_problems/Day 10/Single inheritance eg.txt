import java.util.*;
class Animal {
    void eat(){
        System.out.println("Animal is Eating");
    }
}
class Cat extends Animal{
     void Sound(){
        System.out.println("Cat Meows");
     }
}
public class Main {
     public static void main(String[] args){
        Cat c = new Cat();
        c.eat();
        c.Sound();
     }
}
