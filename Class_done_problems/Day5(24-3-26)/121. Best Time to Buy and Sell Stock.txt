121. Best Time to Buy and Sell Stock
class Solution {
    public int maxProfit(int[] prices) {
         int profit = 0;
         int buy = prices[0];
         for(int i = 1; i < prices.length; i++){
            if(prices[i] < buy){
                  buy = prices[i];
            }
            int curprofit = prices[i] - buy;
            profit = Math.max(profit, curprofit);
         }
         return profit;
    }
}
