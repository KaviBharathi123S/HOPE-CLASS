 import java.util.*;
 public class Print_Without_Space{
 public static void main(String[] args){
        Scanner sc = new Scanner(System.in);
        String s = "";
        String str =  sc.nextLine();
        int len = str.length();
        for(int i = 0; i < len ; i++){
            char ch = str.charAt(i);
            if(ch != ' '){
                s = s + ch;
            }
        }
      System.out.print(s);
 }
}
