import java.util.*;
public class reverse_String1 {
    public static void main(String[] args){
        Scanner sc = new Scanner(System.in);
        String s = "";
        String str =  sc.nextLine();
        int len = str.length();
        for(int i = len - 1; i >= 0; i--){
            char ch = str.charAt(i);
            s = s + ch;
        }
      System.out.print(s);
      sc.close();
    }
}
